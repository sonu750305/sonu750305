void quick_sort(int a[], int low, int high)
{
int j;
if(low<high)
{
j = partition(a, low, high);
quick_sort(a, low, j-1);
quick_sort(a, j+1, high);
}
}
int partition(int a[ ], int low, int high)
{
int i, j, key, temp;
key = a[low];
i = low+1;
j = high;
while(1)
{
while(a[i] < key && i <= high)
{
++i;



}
while(a[j] > key && j >= low)
{
--j;
}
if(i < j)
{
temp = a[i];
a[i] = a[j];
a[j] = temp;
}
else
{
temp = a[low];
a[low] = a[j];
a[j] = temp;
return j;
}
}
